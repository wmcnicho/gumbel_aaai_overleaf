\begin{algorithm*}
\caption{Gumbel Recovery and Counterfactual Generation}
\label{alg:gumbel-cf-long}
\begin{algorithmic}[1]

\STATE \textbf{def} RecoverGumbels$(\mathbf x, \mathbf y, \theta)$:
\STATE \quad $\mathbf G \gets \varnothing$
\STATE \quad \textbf{for} $t=1,\dots,|\mathbf y|$: 
\STATE \qquad $\ell \gets \mathrm{LLM}(\mathbf x, \mathbf y_{<t};\theta)$ 
        \algcmt{compute logits}
\STATE \qquad $m \gets \max_{v\in V} \ell(v)$
        \algcmt{maximum value}
\STATE \qquad sample $g_t(y_t) \sim \mathrm{LowerTruncGumbel}(m - \ell(y_t))$
        \algcmt{recover noise for observed token}
\STATE \qquad $T_t \gets \ell(y_t) + g_t(y_t)$
\STATE \qquad \textbf{for each} $v \in V \setminus \{y_t\}$:
\STATE \qquad\quad sample $g_t(v) \sim \mathrm{UpperTruncGumbel}(T_t - \ell(v))$
        \algcmt{recover shifted noise}
\STATE \qquad store $\mathbf g_t$ in $\mathbf G$
\STATE \quad \textbf{return} $\mathbf G$
\vspace{6pt}

\STATE \textbf{def} GenerateCF$(\mathbf x', \mathbf G, \mathbf y, \theta', \beta)$:
\STATE \quad $\mathbf y' \gets \varnothing$
\STATE \quad $y'_1 \gets \mathrm{BOS},\; t \gets 1$
\STATE \quad \textbf{while} $y'_t \neq \mathrm{EOS}$:
\STATE \qquad $\ell \gets \mathrm{LLM}(\mathbf x', \mathbf y'_{<t};\theta')$ 
        \algcmt{logits under intervened model}
\STATE \qquad \textbf{if} $t \le |\mathbf y|$:
\STATE \qquad\quad $\mathbf g_t \gets \mathbf G[t]$ 
        \algcmt{reuse recovered noise}
\STATE \qquad\quad $y'_t \gets \arg\max_{v\in V}(\ell(v) + \beta g_t(v))$
        \algcmt{noise counterfactual step}
\STATE \qquad \textbf{else}:
\STATE \qquad\quad sample $g_t(v) \sim \mathrm{Gumbel}(0,1)$ 
        \algcmt{generate standard noise if length exceeds original}
\STATE \qquad\quad $y'_t \gets \arg\max_{v\in V}(\ell(v) + u_t(v))$
\STATE \qquad append $y'_t$ to $\mathbf y'$,  $t \gets t+1$
\STATE \quad \textbf{return} $\mathbf y'$
\vspace{6pt}

\STATE \textbf{def} BetaHindsightGenerateCF$(\mathbf x, \mathbf y, \mathbf x', \theta, \theta', \beta)$:
\STATE \quad $\mathbf G \gets$ RecoverGumbels$(\mathbf x, \mathbf y, \theta)$
        \algcmt{recover posterior noise}
\STATE \quad $\mathbf y' \gets$ GenerateCF$(\mathbf x', \mathbf G, \mathbf y, \theta', \beta)$
        \algcmt{construct counterfactual}
\STATE \quad \textbf{return} $\mathbf y'$

\end{algorithmic}
\end{algorithm*}
