\begin{algorithm}
\caption{Student Gumbel Noise Recovery and Counterfactual Generation \mh{TODO 1. need to update inputs, e.g. to add intervention,  prompt components, $\beta $defined. 2. handle case where len(CF)>len(F) 3. Add comments 4. Define TruncGumbels}}
\begin{algorithmic}[1]
\REQUIRE Exercise prompt $E$, Scoring context $C$, Student answer $A=\{a_0,\dots,a_N\}$, Model parameters $\theta$, Vocabulary $V$
\ENSURE Counterfactual answer $A'$
\STATE $G \gets \{\}$ 
\FOR{each token $a_i$ in $A$}
    \STATE $\{l_v\}_{v\in V} = \text{LLM}(E,C,a_0 \dots a_{i-1};\theta)$
    \STATE $l_{\max} \gets \max_{v\in V} l_v$
    \STATE  $g_{a_i} \sim \text{LowerTruncGumbel}(l_{\max}-l_{a_i})$
    \STATE $T \gets l_{a_i} + g_{a_i}$
    \FOR{each $v \in V, v \neq a_i$}
        \STATE  $g_v \sim \text{UpperTruncGumbel}(T-l_v)$
    \ENDFOR
    \STATE Store $g$-vector for position $i$ in $G$
\ENDFOR
\STATE $A' \gets \{\}$
\STATE $a'_i \gets \text{BOS}$
\STATE $i \gets 0$
\WHILE{$a'_i \neq \text{EOS}$}
\STATE $\{l_v\}_{v\in V} = \text{LLM}(E,C',a'_0\dots a'_{i-1};\theta)$
    \IF{$i \le |A|$}
    \STATE $g \gets G[i]$
    \STATE $a'_i \gets \arg\max_{v\in V} (l_v + \beta g_v)$
    \STATE $i \gets i + 1$
    \ELSE
    \STATE $g \gets \text{Gumbel}(0,1)$
    \STATE $a'_i \gets \arg\max_{v\in V} (l_v + g_v)$
    % \STATE $a'_i \gets \arg\max_{v\in V} (l_v)$

    \ENDIF
    \STATE Append $a'_i$ to $A'$
\ENDWHILE
\RETURN $A'$
\end{algorithmic}
\end{algorithm}
